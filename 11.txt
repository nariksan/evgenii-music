        function fmtTime(sec) { if (!sec || isNaN(sec)) return '0:00'; const m = Math.floor(sec / 60), s = Math.floor(sec % 60); return `${m}:${s.toString().padStart(2, '0')}`; }
        function showPlayer() { globalPlayer.classList.add('visible'); }
        function highlightActiveTrack(trackId) {
            document.querySelectorAll('.track-card.active, .compact-item.active').forEach(el => el.classList.remove('active'));
            const activeCard = document.querySelector(`.track-card[data-id="${trackId}"], .compact-item[data-id="${trackId}"]`);
            if (activeCard) activeCard.classList.add('active');
        }
        function loadTrack(index) {
            if (index < 0 || index >= tracks.length) return;
            const track = tracks[index]; currentTrackIndex = index;
            audio.src = track.file; audio.volume = volumeSlider.value;
            playerTitle.textContent = track.title;
            if (track.cover) playerCover.innerHTML = `<img src="${track.cover}" alt="${track.title}">`; else playerCover.textContent = '';
            highlightActiveTrack(track.id); showPlayer();
            audio.play().then(() => setPlayingState(true)).catch(() => setPlayingState(false));
        }
        function setPlayingState(playing) { isPlaying = playing; playerPlayIcon.style.display = playing ? 'none' : 'block'; playerPauseIcon.style.display = playing ? 'block' : 'none'; }
        playPauseBtn.addEventListener('click', () => { if (currentTrackIndex === -1) loadTrack(0); else isPlaying ? audio.pause() : audio.play(); setPlayingState(!isPlaying); });
        prevBtn.addEventListener('click', () => loadTrack(currentTrackIndex > 0 ? currentTrackIndex - 1 : tracks.length - 1));
        nextBtn.addEventListener('click', () => loadTrack(currentTrackIndex < tracks.length - 1 ? currentTrackIndex + 1 : 0));
        audio.addEventListener('timeupdate', () => { if (audio.duration) { playerProgressFill.style.width = `${(audio.currentTime / audio.duration) * 100}%`; playerCurrentTime.textContent = fmtTime(audio.currentTime); } });
        audio.addEventListener('loadedmetadata', () => playerDuration.textContent = fmtTime(audio.duration));
        audio.addEventListener('ended', () => loadTrack(currentTrackIndex < tracks.length - 1 ? currentTrackIndex + 1 : 0));
        playerProgressBar.addEventListener('click', (e) => { if (audio.duration) { const rect = playerProgressBar.getBoundingClientRect(); audio.currentTime = ((e.clientX - rect.left) / rect.width) * audio.duration; } });
        volumeSlider.addEventListener('input', () => audio.volume = volumeSlider.value);
        window.playFromCard = function(index) { loadTrack(index); };

        // RENDERING
        function getCoverHTML(track) { return track.cover ? `<div class="track-cover"><img src="${track.cover}" alt="${track.title}" onerror="this.parentElement.innerHTML='🎵'"></div>` : `<div class="track-cover"></div>`; }
        const downloadIcon = `<svg viewBox="0 0 24 24"><path d="M19 9h-4V3H9v6H5l7 7 7-7zM5 18v2h14v-2H5z"/></svg>`;
        function createFullCard(track) {
            const idx = tracks.findIndex(t => t.id === track.id);
            const card = document.createElement('div'); card.className = 'track-card'; card.dataset.id = track.id;
            card.innerHTML = `${getCoverHTML(track)}<div class="track-player-row"><button class="play-btn" onclick="playFromCard(${idx})"><svg viewBox="0 0 24 24"><polygon points="5,3 19,12 5,21"/></svg></button><div class="track-info"><h3>${track.title}</h3><p>${track.desc || 'Без описания'}</p></div><div style="display:flex;align-items:center;gap:4px;"><span class="track-duration">${track.duration}</span><a href="${track.file}" class="download-btn download-trigger" data-track="${track.title}">${downloadIcon}</a></div></div><button class="release-info-btn" onclick="openReleaseModal('${track.id===1?'modalTrack1':'modalTrack2'}')"><i class="fas fa-info-circle"></i> О релизе</button>`;
            return card;
        }
        function createCompactItem(track) {
            const idx = tracks.findIndex(t => t.id === track.id);
            const item = document.createElement('div'); item.className = 'compact-item'; item.dataset.id = track.id;
            item.innerHTML = `${track.cover ? `<div class="compact-cover"><img src="${track.cover}" alt="${track.title}" onerror="this.parentElement.innerHTML='🎵'"></div>` : `<div class="compact-cover"></div>`}<button class="play-btn compact-play-btn" onclick="playFromCard(${idx})" style="width:36px;height:36px;border-radius:50%;background:linear-gradient(135deg,rgba(255,255,255,0.15),rgba(255,255,255,0.05));color:var(--text-primary);border:1px solid rgba(255,255,255,0.2);display:flex;align-items:center;justify-content:center;cursor:pointer;flex-shrink:0;"><svg viewBox="0 0 24 24" style="width:14px;fill:currentColor;margin-left:2px;"><polygon points="5,3 19,12 5,21"/></svg></button><div class="compact-info"><h4>${track.title}</h4><p>${track.desc || 'Без описания'}</p></div><span class="compact-duration">${track.duration}</span><a href="${track.file}" class="download-btn download-trigger" data-track="${track.title}">${downloadIcon}</a>`;
            return item;
        }
        function renderTracks() {
            const newContainer = document.getElementById('releasesList'), archiveList = document.getElementById('archiveList'), hiddenTracks = document.getElementById('hiddenTracks'), showAllBtn = document.getElementById('showAllBtn');
            const archiveTracks = tracks.filter(t => !t.isNew);
            tracks.filter(t => t.isNew).forEach(track => newContainer.appendChild(createFullCard(track)));
            if (archiveTracks.length > 0) {
                archiveTracks.slice(0, 4).forEach(track => archiveList.appendChild(createCompactItem(track)));
                archiveTracks.slice(4).forEach(track => hiddenTracks.appendChild(createCompactItem(track)));
                showAllBtn.style.display = archiveTracks.length > 4 ? 'flex' : 'none';
            } else { archiveList.innerHTML = '<p style="color:var(--text-secondary);text-align:center;padding:30px;">В архиве пока пусто. Скоро будет 🔜</p>'; showAllBtn.style.display = 'none'; }
            initDownloadButtons();
        }

            const expanded = list.classList.contains('visible'); btn.innerHTML = expanded ? 'Свернуть <span style="transform:rotate(180deg);display:inline-block;">▼</span>' : 'Показать все треки <span>▼</span>'; btn.style.borderColor = expanded ? 'var(--cosmic-gold)' : 'rgba(255,255,255,0.18)';
        }

        const chatToggle = document.getElementById('chatToggle'), chatWindow = document.getElementById('chatWindow'), chatClose = document.getElementById('chatClose'), chatInput = document.getElementById('chatInput'), chatSend = document.getElementById('chatSend'), chatMessages = document.getElementById('chatMessages'), chatClear = document.getElementById('chatClear'), chatStatus = document.getElementById('chatStatus');
        function formatTime(date) { return new Date(date).toLocaleTimeString('ru-RU', { hour: '2-digit', minute: '2-digit' }); }
        function renderMessages(messages) {
            if (!messages || Object.keys(messages).length === 0) { chatMessages.innerHTML = '<div class="empty-chat">Сообщений пока нет.<br>Напиши первое!</div>'; return; }
            chatMessages.innerHTML = ''; Object.values(messages).forEach(msg => { const div = document.createElement('div'), isMe = msg.userId === currentUserId; div.className = `message ${isMe ? 'outgoing' : 'incoming'}`; div.innerHTML = `${msg.text.replace(/[<>]/g, '')}<div class="message-time">${formatTime(msg.timestamp)}</div><div class="message-avatar">${isMe ? 'Я' : (msg.userName || 'Гость')}</div>`; chatMessages.appendChild(div); }); chatMessages.scrollTop = chatMessages.scrollHeight;
        }
        function addMessage(text) { if (!text.trim() || !CHAT_REF) return; CHAT_REF.push({ text: text.trim(), userId: currentUserId, userName: currentUserName || 'Гость', timestamp: Date.now() }); chatInput.value = ''; }
        function clearChat() { const p = prompt('Пароль администратора:'); if (p === '0770' && CHAT_REF) { if (confirm('Очистить чат?')) CHAT_REF.remove(); } else if (p !== null) alert('❌ Неверный пароль!'); }
        function toggleChat() { if (!currentUserName) { const n = prompt('Ваше имя:', 'Гость'); if (n === null) return; currentUserName = n.trim() || 'Гость'; localStorage.setItem('evgenii_user_name', currentUserName); } const v = chatWindow.classList.toggle('visible'); chatToggle.classList.toggle('active', v); chatToggle.textContent = v ? '✕' : '💬'; if (v) setTimeout(() => chatInput.focus(), 100); }
        if (CHAT_REF) { CHAT_REF.on('value', (s) => { renderMessages(s.val()); chatStatus.textContent = '✅ Онлайн'; chatStatus.style.color = 'var(--cosmic-gold)'; }, () => { chatStatus.textContent = '❌ Ошибка'; }); }
        chatToggle.addEventListener('click', toggleChat); chatClose.addEventListener('click', () => { chatWindow.classList.remove('visible'); chatToggle.classList.remove('active'); chatToggle.textContent = '💬'; }); chatSend.addEventListener('click', () => addMessage(chatInput.value)); chatInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') addMessage(chatInput.value); }); chatClear.addEventListener('click', clearChat);
        document.addEventListener('click', (e) => { if (chatWindow.classList.contains('visible') && !chatWindow.contains(e.target) && !chatToggle.contains(e.target)) { chatWindow.classList.remove('visible'); chatToggle.classList.remove('active'); chatToggle.textContent = '💬'; } });
